5/* GUI BUTTONS
6 * BWS = button width (inverse) scaler
7 * BHS = button height (inverse) scaler
8 */
9#define BWS (1.0 / 3.0) // must be between 0 and 1
10#define BHS (1.0 / 5.0) // must be between 0 and 1
11
12/* GUI SAMLPLE ITEMS
13 * SB = sample background scaler
14 * SBS = sample background size
15 * SBX = sample background x
16 * TSBY = top sample background y
17 * BSBY = bottom sample background y
18 */
19#define SB (9.0 / 10.0) // must be between 0 and 1
20#define SBS(mW) ((mW) * SB)
21
22#define SBX(mW) ((mW) * (1 - SB) / 2.0)
23/* mw * bhs + // button width
24 * mw * (1 - 2 * bws) / 1.5 + // two button paddings (top and bot)
25 * mw * 2 * sb + // two sample backgrounds
26 * mw * (1 - sb) / 2 + // one sample background padding
27 *
28 * mw * bhs + mw * (1 - 2 * bws) / 1.5 + mw * 2 * sb + mw * (1 - sb) / 2
29 * mw * (bhs + (1 - 2 * bws) / 1.5 + 2 * sb + (1 - sb) / 2)
30 * mw * (bhs + (1 - 2 * bws) / 1.5 + (3 * sb + 1) / 2)
31 */
32#define TSBY(mW, mH) ((mH) - (mW) * (BHS + (1 - 2 * BWS) / 1.5 + (3 * SB + 1) / 2))
33/* mw * bhs + // button width
34 * mw * (1 - 2 * bws) / 1.5 + // two button paddings (top and bot)
35 * mw * sb + // one sample background
36 *
37 * mw * bhs + mw * (1 - 2 * bws) / 1.5 + mw * sb
38 * mw * (bhs + (1 - 2 * bws) / 1.5 + sb)
39 */
40#define BSBY(mW, mH) ((mH) - (mW) * (BHS + (1 - 2 * BWS) / 1.5 + SB))
41#define TOP_SAMPLE_BACKGROUND_RECT(mW, mH) QRectF(SBX(mW), TSBY(mW, mH), SBS(mW), SBS(mW))
42#define BOT_SAMPLE_BACKGROUND_RECT(mW, mH) QRectF(SBX(mW), BSBY(mW, mH), SBS(mW), SBS(mW))
